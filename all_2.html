go 网络服务

package main

import (
    "net/http"
    "fmt"
)

func test(w http.ResponseWrite,r *http.Request) {
    fmt.Fprintf(w, "hello world %s", r.URL.Path[1:])
}

func main() {
    http.HandleFunc("/", test)
    http.ListenAndServe(":8080", nil)
}
