查看表占用磁盘空间大小

2019/4/4

SELECT table, formatReadableSize(size) as size, rows, days, formatReadableSize(avgDaySize) as avgDaySize FROM (
    SELECT
        table,
        sum(bytes) AS size,
        sum(rows) AS rows,
        min(min_date) AS min_date,
        max(max_date) AS max_date,
        (max_date - min_date) AS days,
        size / (max_date - min_date) AS avgDaySize
    FROM system.parts
    WHERE active
    GROUP BY table
    ORDER BY rows DESC
)

查询最近一段时间内不活跃用户最后的上报page页面人数占比

2019/4/3

SELECT
    cur,
    uniq(uuid) AS u
FROM
(
    SELECT
        uuid,
        cur,
        timestamp
    FROM source.report_data_page
    WHERE ((day >= '2019-03-11') AND (day <= '2019-03-31')) AND (pgac = 0) AND (uuid GLOBAL IN
    (
        SELECT uuid
        FROM
        (
            SELECT
                uuid,
                uniq(day) AS ds
            FROM view.active_users
            WHERE (day >= '2019-03-11') AND (day <= '2019-03-31')
            GROUP BY uuid
            HAVING ds < 3
        )
    ))
    GROUP BY
        uuid,
        cur,
        timestamp
    ORDER BY timestamp DESC
    LIMIT 1 BY timestamp
)
GROUP BY cur
